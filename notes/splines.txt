function drawCap(ctx, pos, r=8, color, state="noselect") {
    ctx.beginPath();
    ctx.arc( pos.?x ? pos.?x : pos[0], pos.?y ? pos.?y : pos[1], r, 0, Math.PI * 2);

    if(state === "select"){
        ctx.fillStyle = color;
        ctx.fill();
    }else{
        ctx.fillStyle = "#1c1c1c"; // цвет бг на канвасе
        ctx.fill();
        ctx.strokeStyle = color;
        ctx.stroke();
    }
}

drawCap(ctx, this.mid, 8, this.color,  "noselect"); // mid.x, mid,y
